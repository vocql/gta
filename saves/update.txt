<script>
/* MENU */
const menuBtn = document.getElementById('menuBtn');
const sideMenu = document.getElementById('sideMenu');
menuBtn.onclick = (e) => { e.stopPropagation(); sideMenu.classList.toggle('active'); };
document.onclick = (e) => { if (!sideMenu.contains(e.target) && e.target !== menuBtn) sideMenu.classList.remove('active'); };

/* HUD DATA & RNG */
let currentMoney = 19052025;
const locations = ["VICE CITY", "PORT GELLHORN", "LEONIDA", "AMBROSIA"];
const moneyEl = document.getElementById('hudMoney');
const locEl = document.getElementById('locDisplay');
const starContainer = document.getElementById('starContainer');

function updateHUD() {
  const now = new Date();
  document.getElementById('hudTime').textContent = `${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}`;
  
  currentMoney += Math.floor(Math.random() * 495) + 5;
  moneyEl.textContent = `$${currentMoney.toLocaleString()}`;
  moneyEl.style.transform = 'scale(1.05)';
  setTimeout(() => moneyEl.style.transform = 'scale(1)', 100);
}

function updateStars() {
  // Random Star Level
  const level = Math.floor(Math.random() * 7); 
  const stars = starContainer.querySelectorAll('.star');
  stars.forEach((s, i) => i < level ? s.classList.add('active') : s.classList.remove('active'));
  
  // Random Location Swap
  locEl.textContent = locations[Math.floor(Math.random() * locations.length)];
}

setInterval(updateHUD, 2000);
setInterval(updateStars, 60000); 
updateHUD(); updateStars();

/* RELEASE COUNTDOWN */
const targetDate = new Date('2026-11-19T00:00:00');
function updateCountdown(){
  const diff = Math.floor((targetDate - new Date())/1000);
  if(diff<=0) return;
  document.getElementById('days').textContent = Math.floor(diff/86400);
  document.getElementById('hours').textContent = String(Math.floor(diff/3600%24)).padStart(2,'0');
  document.getElementById('minutes').textContent = String(Math.floor(diff/60%60)).padStart(2,'0');
  document.getElementById('seconds').textContent = String(diff%60).padStart(2,'0');
}
setInterval(updateCountdown,1000);
updateCountdown();

/* MUSIC */
const musicTrigger = document.getElementById('musicTrigger');
const wheel = document.getElementById('musicWheel');
const player = document.getElementById('musicPlayer');
musicTrigger.onclick = () => wheel.classList.toggle('active');
document.querySelectorAll('.music-item').forEach(btn=>{
  btn.onclick = () => { player.src = btn.dataset.song; player.play(); };
});
</script>
